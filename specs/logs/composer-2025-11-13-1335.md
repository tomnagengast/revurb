      ): Record<string, string> | null {
        // Convert pattern to regex: /apps/{appId}/events => /apps\/([^/]+)\/events/
        const regexPattern = pattern
          .replace(/\//g, "\\/")
          .replace(/\{([^}]+)\}/g, "([^/]+)");
    
        const regex = new RegExp(`^${regexPattern}$`);
        const match = pathname.match(regex);
